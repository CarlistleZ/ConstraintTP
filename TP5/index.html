vecProduct(V1, V2, VRes):- 
    dim(V1,[D]),
    dim(V2,[D]),
    dim(VRes,[D]),
    (foreacharg(Elt1,V1), foreacharg(Elt2,V2), foreacharg(EltRes,VRes) do
        EltRes #= Elt1 * Elt2
    ).

vecSum(Vec,Sum):-(
    foreacharg(Elt,Vec), fromto(0,In,Out,Sum) do
        Out #= In + Elt
    ).

vecDotProduct(V1, V2, S):-
    vecProduct(V1, V2, TmpVec),
    vecSum(TmpVec, S).
